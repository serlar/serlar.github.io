GrB_Info GrB_Matrix_CSC_or_CSR_extractElement_##T(type *p, \
GrB_Matrix m_in, \
GrB_Index row, \
GrB_Index col) \
{ \
if (m_in->element_type != GrB_##T) \
GrB_error("The data types of the matrix elements and the \ target do not match in GrB_Matrix_CSC_or_CSR_extractElement!\n"); \
GrB_Matrix_CSC_or_CSR *m = (GrB_Matrix_CSC_or_CSR *)(m_in); \
\
GrB_Index index1 = m_in->format_type == GrB_CSC ? col : row; \
GrB_Index index2 = m_in->format_type == GrB_CSC ? row : col; \
\
GrB_Index ptr_start = m->ptr[index1]; \
GrB_Index n_one_line = m->ptr[index1 + 1] - ptr_start; \
\
for (GrB_Index i = 0; i < n_one_line; ++i) \
{ \
GrB_Index index2_temp = m->indices[i + ptr_start]; \
if (index2 == index2_temp) \
{ \
*p = ((type *)m->val)[i + ptr_start]; \
return GrB_SUCCESS; \
} \
else if (index2 < index2_temp) \
break; \
} \
return GrB_NO_VALUE; \
}
